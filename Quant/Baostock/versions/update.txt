from common import *

def get_latest_dates(ch_client=None):
    """
    获取每只股票的最新日期
    :param ch_client: 可选的ClickHouse连接对象
    :return: 字典 {股票代码: 最新日期字符串}
    """
    close_connection = False
    try:
        # 如果没有传入连接，则创建新连接
        if ch_client is None:
            ch_client = get_clickhouse_client()
            close_connection = True
        
        # 执行查询
        result = ch_client.execute("""
        SELECT 
            code, 
            toString(max(date)) as max_date
        FROM daily_k
        GROUP BY code
        """)
        
        return {code: max_date for code, max_date in result}
        
    finally:
        # 如果是自己创建的连接，则关闭它
        if close_connection and ch_client:
            ch_client.disconnect()

def fetch_update_data(stock_codes):
    """终极稳定版数据更新函数"""
    # 配置
    CH_POOL_SIZE = 5
    MAX_WORKERS = 15
    BATCH_SIZE = 50
    REQUEST_INTERVAL = 0.5  # 秒
    
    # 初始化
    ch_pool = [get_clickhouse_client() for _ in range(CH_POOL_SIZE)]
    bs_manager = BaostockManager()
    
    try:
        # 获取最新日期
        with ch_pool[0] as ch:
            latest_dates = get_latest_dates(ch)
        end_date = datetime.now().date()
        
        # 分批处理
        success_count = 0
        for batch_idx in range(0, len(stock_codes), BATCH_SIZE):
            batch_codes = stock_codes[batch_idx:batch_idx+BATCH_SIZE]
            
            with concurrent.futures.ThreadPoolExecutor(
                max_workers=min(MAX_WORKERS, len(batch_codes)),
                thread_name_prefix="stock_"
            ) as executor:
                futures = {}
                
                for code in batch_codes:
                    try:
                        # 计算日期范围
                        last_date_str = latest_dates.get(code)
                        start_date = (datetime.now() - timedelta(days=7)).date()
                        
                        if last_date_str:
                            try:
                                last_date = datetime.strptime(last_date_str, "%Y-%m-%d").date()
                                start_date = last_date + timedelta(days=1)
                            except Exception as e:
                                logger.warning(f"股票 {code} 日期解析失败: {str(e)}")
                        
                        if start_date > end_date:
                            logger.debug(f"股票 {code} 数据已是最新")
                            continue
                            
                        # 提交任务
                        future = executor.submit(
                            fetch_stock_data,
                            code=code,
                            start_date=start_date.strftime("%Y-%m-%d"),
                            end_date=end_date.strftime("%Y-%m-%d")
                        )
                        futures[future] = code
                        time.sleep(REQUEST_INTERVAL)  # 控制请求频率
                    except Exception as e:
                        logger.error(f"股票 {code} 任务提交失败: {str(e)}")
                
                # 处理结果
                for future in concurrent.futures.as_completed(futures):
                    code = futures[future]
                    try:
                        df = future.result()
                        if not df.empty:
                            ch = ch_pool[hash(code) % CH_POOL_SIZE]
                            bulk_insert_to_clickhouse(ch, df)
                            success_count += 1
                    except Exception as e:
                        logger.error(f"股票 {code} 处理失败: {str(e)}")
            
            logger.info(f"批次 {batch_idx//BATCH_SIZE + 1} 完成，成功率: {success_count}/{len(batch_codes)}")
            
        logger.info(f"全部完成，总计成功: {success_count}/{len(stock_codes)}")
        
    finally:
        # 清理资源
        for conn in ch_pool:
            try:
                conn.disconnect()
            except:
                pass
        try:
            bs.logout()
        except:
            pass
    
if __name__ == "__main__":
    logger.info("===== 开始每日数据更新 =====")
    stock_codes = get_all_stock_codes()
    logger.info(f"共获取 {len(stock_codes)} 只股票")
    
    fetch_update_data(stock_codes)
    logger.info("===== 每日数据更新完成 =====")
